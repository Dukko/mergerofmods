can_resettle_pop = {
    custom_tooltip = {
        fail_text = "self_modified_refuse"
        NOT = {
            has_trait = trait_self_modified
        }
    }
    IF = {
        limit = {
            OR = {
                has_trait = trait_living_architecture
                has_trait = trait_rooted_adults_rooted
            }
        }
        always = no
    }
    ELSE_IF = {
        limit = {
            is_shackled_robot = yes
        }
        always = yes
    }
    ELSE_IF = {
        limit = {
            planet = {
                OR = {
                    has_modifier = doomsday_1
                    has_modifier = doomsday_2
                    has_modifier = doomsday_3
                    has_modifier = doomsday_4
                    has_modifier = doomsday_5
                }
            }
        }
        always = yes
    }
    ELSE_IF = {
        limit = {
            planet = {
                has_modifier = planet_culture_shock
            }
        }
        OR = {
            is_same_species = planet.owner
            planet = {
                NOT = {
                    has_modifier = planet_culture_shock
                }
            }
        }
    }
    else_if = {
        limit = {
            owner = {
                is_gestalt = yes
            }
            OR = {
                NOR = {
                    has_policy_flag = resettlement_not_allowed
                    has_policy_flag = resettlement_allowed
                }
                drone_job_check_trigger = yes
            }
        }
        always = yes
    }
    else = {
        custom_tooltip = {
            fail_text = RESETTLEMENT_POLICY_FAIL
            owner = {
                has_policy_flag = resettlement_allowed
            }
        }
    }
}
