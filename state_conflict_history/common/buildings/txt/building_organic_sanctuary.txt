building_organic_sanctuary = {
    base_buildtime = @b1_time
    category = unity
    potential = {
        exists = owner
        owner = {
            has_valid_civic = civic_machine_servitor
        }
    }
    allow = {
        exists = owner
        OR = {
            owner = {
                is_ai = no
            }
            AND = {
                num_buildings = {
                    type = building_organic_sanctuary
                    value < 2
                }
                NOT = {
                    has_building = building_organic_paradise
                }
            }
        }
    }
    destroy_trigger = {
        exists = owner
        OR = {
            owner = {
                NOT = {
                    owner = {
                        has_valid_civic = civic_machine_servitor
                    }
                }
            }
            AND = {
                owner = {
                    is_ai = yes
                }
                count_owned_pop = {
                    limit = {
                        has_citizenship_type = {
                            type = citizenship_organic_trophy
                        }
                    }
                    count <= 20
                }
                num_buildings = {
                    type = building_organic_sanctuary
                    value > 2
                }
            }
            AND = {
                owner = {
                    is_ai = yes
                }
                count_owned_pop = {
                    limit = {
                        has_citizenship_type = {
                            type = citizenship_organic_trophy
                        }
                    }
                    count <= 40
                }
                num_buildings = {
                    type = building_organic_paradise
                    value > 3
                }
            }
        }
    }
    convert_to = {
        building_luxury_residence
        building_communal_housing
    }
    planet_modifier = {
        job_bio_trophy_add = 10
        job_artisan_drone_add = 1
    }
    resources = {
        category = planet_buildings
        cost = {
            minerals = @b1_minerals
        }
        upkeep = {
            energy = @b1_upkeep
        }
    }
    upgrades = {
        "building_organic_paradise"
    }
    ai_weight = {
        weight = 0
        modifier = {
            weight = 5000
            any_owned_pop = {
                is_unemployed = yes
                has_citizenship_type = {
                    type = citizenship_organic_trophy
                }
            }
        }
        modifier = {
            weight = 5000
            NOR = {
                has_building = building_organic_sanctuary
                has_building = building_organic_paradise
            }
            planet = {
                owner = {
                    any_owned_pop = {
                        is_organic_species = yes
                        prevprev = {
                            habitability = {
                                who = prev
                                value > 0.5
                            }
                        }
                    }
                }
            }
        }
    }
}
