namespace = action
country_event = {
    id = action.7
    hide_window = yes
    is_triggered_only = yes
    trigger = {
        is_country_type = default
        is_hostile = from
    }
    immediate = {
        if = {
            limit = {
                num_killed_ships = {
                    target = from
                    value < 10
                }
                num_taken_planets = {
                    target = from
                    value = 0
                }
            }
            add_opinion_modifier = {
                modifier = opinion_first_contact_hostility
                who = from
            }
        }
        if = {
            limit = {
                OR = {
                    num_killed_ships = {
                        target = from
                        value > 9
                    }
                    num_taken_planets = {
                        target = from
                        value > 0
                    }
                }
            }
            add_opinion_modifier = {
                modifier = opinion_first_contact_war
                who = from
            }
            set_country_flag = shoot_to_kill_achievement
        }
    }
}
