namespace = action
country_event = {
    id = action.80
    hide_window = yes
    is_triggered_only = yes
    trigger = {
        last_changed_policy = robot_pop_policy
        has_policy_flag = robots_outlawed
        any_owned_army = {
            OR = {
                army_type = robotic_army
                army_type = robotic_defense_army
            }
        }
    }
    immediate = {
        if = {
            limit = {
                last_changed_policy = robot_pop_policy
                has_policy_flag = robots_outlawed
            }
            every_owned_army = {
                limit = {
                    OR = {
                        army_type = robotic_army
                        army_type = robotic_defense_army
                    }
                }
                remove_army = yes
            }
        }
    }
}
