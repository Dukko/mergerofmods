building_hive_warren = {
    base_buildtime = @b1_time
    category = amenity
    potential = {
        exists = owner
        owner = {
            is_hive_empire = yes
        }
    }
    allow = {
        buildings_simple_allow = yes
        or = {
            owner = {
                is_ai = no
            }
            free_jobs > -2
            free_amenities < 0
            num_pops > 13
        }
    }
    destroy_trigger = {
        exists = owner
        OR = {
            AND = {
                planet = {
                    free_building_slots = 0
                    free_jobs < -3
                    free_housing > 6
                    free_district_slots = 0
                    owner = {
                        is_ai = yes
                    }
                }
            }
            NOR = {
                owner = {
                    is_hive_empire = yes
                }
            }
        }
    }
    convert_to = {
        building_luxury_residence
        building_drone_storage
        building_communal_housing
    }
    planet_modifier = {
        planet_housing_add = 3
        planet_amenities_add = 5
    }
    resources = {
        category = planet_buildings
        cost = {
            minerals = @b1_minerals
        }
        upkeep = {
            energy = 2
        }
    }
    upgrades = {
        "building_expanded_warren"
        "ebh_hive_warren"
    }
}
