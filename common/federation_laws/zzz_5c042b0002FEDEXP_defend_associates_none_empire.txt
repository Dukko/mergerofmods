    potential = {
        has_federation_type = vf_ef_federation
    }
    allow = {
        custom_tooltip = {
            NOT = {
                AND = {
                    NOT = {
                        has_federation_law = FEDEXP_defend_associates_none_empire
                    }
                    has_modifier = law_category_FEDEXP_defend_associates_cooldown
                    has_any_federation_law_in_category = law_category_FEDEXP_defend_associates
                }
            }
        }
        custom_tooltip = {
            fail_text = EMPIRE_LAW_CHANGE
            NOT = {
                AND = {
                    NOT = {
                        has_federation_law = FEDEXP_defend_associates_none_empire
                    }
                    has_federation_flag = vf_law_changes_disabled
                }
            }
        }
    }
    on_enact = {
        if = {
            limit = {
                has_federation_flag = enable_federation_cooldowns
            }
            add_modifier = {
                modifier = law_category_FEDEXP_defend_associates_cooldown
                days = @cooldown
            }
        }
        add_cohesion = -5
    }
    ai_weight = {
        base = 51
    }
