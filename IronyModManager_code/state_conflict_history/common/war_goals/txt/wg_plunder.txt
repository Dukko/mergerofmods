wg_plunder = {
    casus_belli = cb_despoliation
    hide_if_no_cb = no
    total_war = no
    galactic_empire_joins_defender = {
        from = {
            is_galactic_emperor = yes
        }
        imperial_authority >= 100
    }
    allowed_peace_offers = {
        status_quo
        surrender
        demand_surrender
    }
    surrender_acceptance = -25
    potential = {
        or = {
            is_ai = no
            years_passed > @str_ai_will_not_declare_war_until
        }
        has_total_war_cb = no
        has_valid_civic = civic_barbaric_despoilers
        NOT = {
            is_in_federation_with = from
        }
        NAND = {
            is_galactic_emperor = yes
            from = {
                is_galactic_community_member = yes
            }
        }
    }
    possible = {
        custom_tooltip = {
            fail_text = has_claimed_all_systems
            from = {
                is_country_type = default
                any_owned_planet = {
                    is_colony = yes
                    root = {
                        NOT = {
                            has_claim = prev.solar_system
                        }
                    }
                }
            }
        }
        custom_tooltip = {
            fail_text = cannot_use_against_emperor
            OR = {
                from = {
                    is_galactic_emperor = no
                }
                is_galactic_community_member = no
            }
        }
        custom_tooltip = {
            fail_text = pax_galactica_active
            NAND = {
                is_active_resolution = resolution_emperor_pax_galactica
                is_galactic_community_member = yes
                from = {
                    is_galactic_community_member = yes
                }
            }
        }
    }
    on_accept = {
        custom_tooltip = wg_plunder_accept_them
        add_resource = {
            energy = 10
            minerals = 10
            alloys = 10
            consumer_goods = 10
            mult = from.trigger:num_pops
        }
        if = {
            limit = {
                from = {
                    OR = {
                        has_technology = tech_mine_volatile_motes
                        has_technology = tech_volatile_motes
                    }
                }
            }
            add_resource = {
                volatile_motes = 1
                mult = from.trigger:num_pops
            }
        }
        if = {
            limit = {
                from = {
                    OR = {
                        has_technology = tech_mine_exotic_gases
                        has_technology = tech_exotic_gases
                    }
                }
            }
            add_resource = {
                exotic_gases = 1
                mult = from.trigger:num_pops
            }
        }
        if = {
            limit = {
                from = {
                    OR = {
                        has_technology = tech_mine_rare_crystals
                        has_technology = tech_rare_crystals
                    }
                }
            }
            add_resource = {
                rare_crystals = 1
                mult = from.trigger:num_pops
            }
        }
        hidden_effect = {
            from = {
                add_modifier = {
                    modifier = raided
                    days = 3600
                }
            }
        }
    }
    on_wargoal_set = {
        random_war = {
            limit = {
                using_war_goal = {
                    type = wg_end_threat
                    owner = root
                }
            }
            every_war_participant = {
                limit = {
                    has_country_flag = empire_fed_no_war_subject
                }
                prev = {
                    remove_war_participant = prev
                }
                }
            }
        }
    on_status_quo = {
        add_resource = {
            energy = 500
            minerals = 500
            alloys = 500
            consumer_goods = 500
        }
    }
    ai_weight = {
        weight = 100
    }
}
